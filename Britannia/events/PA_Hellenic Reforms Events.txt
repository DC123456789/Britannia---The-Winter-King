narrative_event = {
	id = HEL.212
	title = EVTNAMEHEL.212
	desc = EVTDESCHEL.212
	picture = GFX_evt_roman_temple
	border = GFX_event_narrative_frame_religion
	
	major = yes
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTAancrel.2104
		ai_chance = {
			factor = 50
			modifier = {
				factor = 5
				any_liege = { religion = ROOT }
			}
			modifier = {
				factor = 2
				same_realm = ROOT
			}
			modifier = {
				factor = 5
				trait = cynical
			}
			modifier = {
				factor = 10
				dynasty = ROOT
			}
			modifier = {
				factor = 0.1
				trait = zealous
			}
			modifier = {
				factor = 0.5
				trait = content
			}
		}
		trigger = {
			OR = {
				religion = sol_invictus
				religion = roman_pagan
			}
		}
		if = {
			limit = { religion = roman_pagan }
			religion = sol_invictus
		}
		hidden_tooltip = {
			any_courtier = {
				limit = {
					is_heresy_of = ROOT
					employer = { character = PREV }
				}
				religion = ROOT
			}
			any_vassal = {
				limit = {
					is_playable = no
					is_heresy_of = ROOT
				}
				religion = ROOT
			}
			capital_scope = {
				if = {
					limit = { is_heresy_of = ROOT }
					religion = ROOT
				}
			}
			any_demesne_title = {
				limit = { holy_order = yes }
				religion = ROOT
			}
			religion = ROOT
		}
	}
	
	option = {
		name = EVTOPTC670
		ai_chance = { factor = 25 }
		trigger = {
			religion = roman_pagan
		}
	}
	
	option = {
		name = EVTOPTBancrel.2104
		trigger = {
			NOR = { 
				religion = sol_invictus 
				religion = roman_pagan
			}
		}
	}
}
