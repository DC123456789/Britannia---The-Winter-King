# You summon your vassals to the divination
character_event = {
	id = ancrel.0020
	desc = EVTDESCancrel.0020
	picture = GFX_evt_viking_throneroom_oldgods
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	hide_from = yes
	
	option = {
		name = EVTOPTAancrel.0020
		any_vassal = {
			limit = {
				OR = {
					religion = briton_pagan
					religion = briton_pagan_reformed
					religion = pict_pagan
					religion = pict_pagan_reformed
					religion = gael_pagan
					religion = gael_pagan_reformed
				}
				age = 15
				prisoner = no
				war = no
				NOT = { trait = incapable }
			}
			character_event = { id = ancrel.0021 tooltip = EVTTOOLTIPancrel.0021 }
			hidden_tooltip = { character_event = { id = ancrel.0025 days = 10 } }
		}
		any_vassal = {
			limit = {
				OR = {
					religion_group = christian
					#religion_group = muslim
				}
				age = 15
				prisoner = no
				war = no
				NOT = { trait = incapable }
			}
			character_event = { id = ancrel.0022 tooltip = EVTTOOLTIPancrel.0022 }
		}
		hidden_tooltip = { character_event = { id = ancrel.0024 days = 10 } }
	}
}

# Non-Druidic Pagans pay to skip divination
character_event = {
	id = ancrel.0022
	desc = EVTDESCancrel.0022
	picture = GFX_evt_viking_throneroom_oldgods
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	hide_from = yes
	
	option = { # Pay
		name = EVTOPTAancrel.0022
		trigger = {
			wealth = 20
		}
		FROM = {
			character_event = { id = ancrel.0023 tooltip = EVTTOOLTIPancrel.0023 }
		}
		wealth = -20
	}
	option = { # Don't pay
		name = EVTOPTBancrel.0022
		FROM = {
			character_event = { id = ancrel.0056 tooltip = EVTTOOLTIPancrel.0056 }
		}
		tooltip = {
			FROM = {
				opinion = {
					modifier = opinion_refused_divination
					who = ROOT
					years = 2
				}
				any_vassal = {
					limit = { has_character_flag = attending_divination }
					opinion = {
						modifier = opinion_refused_divination
						who = ROOT
						years = 2
					}
				}
			}
		}
	}
}
