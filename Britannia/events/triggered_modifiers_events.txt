# Small Demesne
character_event = {
	id = 301906
	hide_window = yes
	
	only_playable = yes
	
	trigger = {
		OR = {
			tier = count
			tier = duke
		}
		NOT = { trait = incapable }
		realm_stewardship = 20
		war = no
		OR = {
			AND = {
				demesne_size = 3
				NOT = { demesne_size = 4 }
				NOT = { has_character_modifier = small_demesne_1 }
			}
			AND = {
				demesne_size = 2
				NOT = { demesne_size = 3 }
				NOT = { has_character_modifier = small_demesne_2 }
			}
			AND = {
				NOT = { demesne_size = 2 }
				NOT = { has_character_modifier = small_demesne_3 }
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		remove_character_modifier = small_demesne_1
		remove_character_modifier = small_demesne_2
		remove_character_modifier = small_demesne_3
		if = {
			limit = {
				NOT = { demesne_size = 2 }
			}
			add_character_modifier = {
				modifier = small_demesne_3
				duration = -1
			}
			break = yes
		}
		if = {
			limit = {
				NOT = { demesne_size = 3 }
			}
			add_character_modifier = {
				modifier = small_demesne_2
				duration = -1
			}
			break = yes
		}
		add_character_modifier = {
			modifier = small_demesne_1
			duration = -1
		}
	}
}

		}
		export_to_variable = {
			which = base_levies_realm_size
			value = realm_size
		}

		if = {
			limit = {
				primary_title = {

				}
			}
			divide_variable = {
				which = base_levies_realm_size
				value = 1.25
			}
		}

			limit = {
				primary_title = {
					has_law = crown_authority_2
				}
			}

				which = base_levies_realm_size
				value = 1.5
			}
		}

				primary_title = {
					has_law = crown_authority_3
