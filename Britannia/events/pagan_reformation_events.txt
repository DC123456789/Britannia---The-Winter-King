character_event = {			# Stop Brythonic Reformation if Old Gods haven't been restored
	id = 316000
	title = "EVTNAME316000"
	desc = "EVTDESC316000"
	picture = GFX_evt_druidic_stone_forest
	border = "GFX_event_normal_frame_religion"
	
	is_triggered_only = yes
	hide_new = yes
	
	trigger = {
		always = no				# Doesn't work
		NOT = { has_global_flag = old_gods_restored }
		religion = briton_pagan_reformed
		NOT = {
			has_alternate_start_parameter = { key = religion_names value = random }
		}
	}
	
	immediate = {
		clr_global_flag = briton_reformation
		activate_title = {
			title = d_briton_pagan
			status = no
		}
		d_briton_pagan = { destroy_landed_title = THIS }
		religion = briton_pagan
		piety = 750
		if = {		# Try to stop the AI spamming reformation
			limit = { 
				ai = yes 
				NOT = { num_of_holy_sites = 5 }
			}
			piety = -100
		}
		set_parent_religion = {
			religion = briton_pagan
			parent = 0
		}
	}

	option = {
		name = "EVTOPTA316000" # Start Ambition
		trigger = {
			age = 15
			religion = briton_pagan	
			NOT = { character = 3 }
			NOT = { trait = incapable }
			NOT = {
				AND = {
					has_character_flag = abandoned_restore_quest
					NOT = { has_character_flag = char_restored_old_gods }
				}
			}
			NOT = { has_global_flag = old_gods_restored }
			OR = {
				NOT = { has_global_flag = ritual_performed }
				had_global_flag = {
					flag = ritual_performed
					days = 3650
				}
			}
			NOT = { has_character_flag = restoring_old_gods }
			prisoner = no
			piety = 300
			NOT = { trait = content }
			NOT = { trait = cynical }
			NOT = { trait = tolerant }
			NOT = { trait = imbecile }
			OR = {
				has_focus = focus_theology
				NOT = { has_dlc = "Way of Life" }
			}
		}
		set_character_flag = restoring_old_gods
		custom_tooltip = {
			text = restore_old_gods_tooltip
		}
		hidden_tooltip = {
			character_event = { id = 3124500 }
		}
	}
	option = {
		name = "EVTOPTB316000"
	}
}

character_event = {				# Set up Brythonic Archdruid title
	id = 316001
	desc = "You're not supposed to see this..."
	picture = GFX_evt_druidic_stone_forest
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		has_global_flag = old_gods_restored
		religion = briton_pagan_reformed
		briton_pagan_reformed = {
			OR = {
				has_religion_features = no
				has_religion_feature = religion_druidic_leadership
			}
		}
	}
	
	immediate = {
		activate_title = {
			title = d_briton_pagan
			status = yes
		}
		any_realm_lord = { 		# Try to give the title to Merlin first
			limit = {
				has_character_flag = merlin_of_avalon
				OR = {
					religion = briton_pagan
					religion = briton_pagan_reformed
				}
				NOT = { trait = incapable }
				OR = {
					trait = druid
					trait = archdruid
				}
			}
			religion = briton_pagan_reformed
			character_event = { id = 316002 }
			break = yes
		}	
		any_independent_ruler = {		
			any_realm_lord = {
				limit = { 
					has_character_flag = merlin_of_avalon
					OR = {
						religion = briton_pagan
						religion = briton_pagan_reformed
					}
					NOT = { trait = incapable }
					OR = {
						trait = druid
						trait = archdruid
					}
				}
				religion = briton_pagan_reformed
				character_event = { id = 316002 }
				break = yes
			}
		}
		if = {				# Reformer gets to keep title if already a druid
			limit = {
				OR = {
					trait = druid
					trait = archdruid
				}
			}
			character_event = { id = 316002 }
			break = yes
		}
		any_realm_lord = { 		# Try to give the title to a suitable archdruid
			limit = {
				OR = {
					religion = briton_pagan
					religion = briton_pagan_reformed
				}
				NOT = { trait = incapable }
				trait = archdruid
			}
			religion = briton_pagan_reformed
			character_event = { id = 316002 }
			break = yes
		}	
		any_independent_ruler = {		
			any_realm_lord = {
				limit = { 
					OR = {
						religion = briton_pagan
						religion = briton_pagan_reformed
					}
					NOT = { trait = incapable }
					trait = archdruid
				}
				religion = briton_pagan_reformed
				character_event = { id = 316002 }
				break = yes
			}
		}
		any_realm_lord = { 		# Try to give the title to a suitable druid
			limit = {
				OR = {
					religion = briton_pagan
					religion = briton_pagan_reformed
				}
				NOT = { trait = incapable }
				trait = druid
			}
			religion = briton_pagan_reformed
			character_event = { id = 316002 }
			break = yes
		}
		any_independent_ruler = {		
			any_realm_lord = {
				limit = { 
					OR = {
						religion = briton_pagan
						religion = briton_pagan_reformed
					}
					NOT = { trait = incapable }
					trait = druid
				}
				religion = briton_pagan_reformed
				character_event = { id = 316002 }
				break = yes
			}
		}			# If all else fails, the title is destroyed	
		character_event = { id = 316004 }
	}
	
	option = {
		name = OK
	}
}

character_event = {			# Bounce events to all candidates
	id = 316013
	desc = "EVTDESC316013"
	picture = GFX_evt_emissary
	border = "GFX_event_normal_frame_religion"
	
	hide_window = yes
	
	only_rulers = yes
	
	trigger = {
		has_global_flag = briton_druid_invites_sent
		NOT = { has_global_flag = no_briton_archdruid_successor }
		has_landed_title = d_briton_pagan
		briton_pagan_reformed = {
			OR = {
				has_religion_features = no
				has_religion_feature = religion_druidic_leadership
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 0.0001
			always = yes
		}
	}
	
	immediate = {
		clr_global_flag = briton_druid_invites_sent
		# Clean up previous event
		top_liege = {
			if = {
				limit = { has_character_flag = briton_invite_received }
				clr_character_flag = briton_invite_received
			}
			any_realm_lord = {
				limit = { has_character_flag = briton_invite_received }
				clr_character_flag = briton_invite_received
			}
		}
		any_independent_ruler = {
			if = {
				limit = { has_character_flag = briton_invite_received }
				clr_character_flag = briton_invite_received
			}		
			any_realm_lord = {
				limit = { has_character_flag = briton_invite_received }
				clr_character_flag = briton_invite_received
			}
		}
		if = {
			limit = {
				d_briton_pagan = { check_variable = { which = "num_druid_candidates" value = 2 } }				# If at least 2 candidates, score and compare them
			}
			top_liege = {
				if = {
					limit = { 
						has_character_flag = briton_druid_candidate

	desc = "EVTDESC316301"

	border = "GFX_event_normal_frame_religion"
	
	is_triggered_only = yes

	option = {						# Accept Pontificate
		name = "EVTOPTA316301"
		prestige = 200
		piety = 100
		d_roman_pagan = {
			usurp_title_only = ROOT
		}
		hidden_tooltip = {
			any_independent_ruler = {
				if = {
					limit = { 
						religion = roman_pagan_reformed 
						NOT = { character = ROOT }
						NOT = { character = FROM }
					}
					character_event = { id = 316302 }
				}
				any_realm_lord = {
					limit = { 
						religion = roman_pagan_reformed 
						NOT = { character = ROOT }
						NOT = { character = FROM }
					}
					character_event = { id = 316302 }
				}
			}
		}
	}

	option = {						# Refuse Pontificate
		name = "EVTOPTB316301"
		prestige = -100
		hidden_tooltip = {
			FROM = {
				if = {
					limit = {
						tier = DUKE
						NOT = { num_of_duke_titles = 2 }
					}
					liege = {
						save_event_target_as = character_liege
						ROOT = {
							any_vassal = {
								limit = { tier = COUNT }
								set_defacto_liege = PREVPREV
							}
							if = { 
								limit = { NOT = { num_of_count_titles = 1 } }
								any_vassal = {
									limit = { tier = BARON }
									set_defacto_liege = PREVPREV
								}
							}
							d_roman_pagan = { destroy_landed_title = THIS }
						}
					}
					event_target:character_liege = { set_defacto_vassal = ROOT }
					clear_event_target = character_liege
				}
			}
		}
		d_roman_pagan = {
			destroy_landed_title = THIS
		}
	}
}
