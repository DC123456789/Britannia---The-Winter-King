##### TRIBAL GOVERNMENT #####
# Small Tribal Kings or Empires are destroyed
character_event = { 
	id = 302500
	title = "EVTNAME302500"
	desc = "EVTDESC302500"
	hide_new = yes
	
	picture = "GFX_evt_riders_on_hill"
	
	only_playable = yes
	only_independent = yes
	
	is_triggered_only = yes
	
	trigger = {
		government = tribal_government
		NOT = { num_of_count_titles_in_realm = 8 }
		NOT = { prestige = 1500 }
		any_demesne_title = {
			higher_tier_than = DUKE
			NAND = {
				tier = KING
				empire = {
					OR = {
						title = e_ireland
						title = e_alba
						title = e_britannia
						title = e_cymru
					}
				}
				ROOT = { culture_group = celtic }
			}
		}
		NAND = {		# Not in a civil war
			war = yes
			any_war = {
				OR = {
					AND = {
						attacker = { character = ROOT }
						defender = {
							in_revolt = yes
						}
					}
					AND = {
						defender = { character = ROOT }
						attacker = {
							in_revolt = yes
						}
					}
				}
			}
		}
	}
	
	immediate = {
		primary_title = {
			save_event_target_as = former_primary_title
		}
	}
	
	option = {
		name = "EVTOPTA302500"
		hidden_tooltip = {
			any_vassal = {
				limit = { 
					defacto_liege_title = {
						holder_scope = { character = ROOT }
						higher_tier_than = DUKE
						NAND = {
							tier = KING
							empire = {
								OR = {
									title = e_ireland
									title = e_alba
									title = e_britannia
									title = e_cymru
								}
							}
							ROOT = { culture_group = celtic }
						}						
					}
				}
				character_event = { id = 302501 }
			}
		}
		any_demesne_title = {
			limit = {
				higher_tier_than = DUKE
				NAND = {
					tier = KING
					empire = {
						NOR = {
							title = e_ireland
							title = e_alba
							title = e_britannia
							title = e_cymru
						}
					}
					ROOT = { culture_group = celtic }
				}
			}
			destroy_landed_title = THIS
		}
	}
}

# When a county changes hands
character_event = {
	id = 302531
	desc = "You're not supposed to see this..."
	hide_window = yes
	
	picture = "GFX_evt_riders_on_hill"
	
	is_triggered_only = yes
	
	immediate = {
		FROM = {
			if = {
				limit = {
					tier = COUNT
				}
				location = {
					province_event = { id = 302532 }
				}
			}
		}
	}
}

# Actual event for province
province_event = {
	id = 302532
	desc = "You're not supposed to see this..."
	hide_window = yes
	
	picture = "GFX_evt_riders_on_hill"
	
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = {
				owner = {
					top_liege = {
						OR = {
							government = imperial_government
							government = sub_roman_government
							government = germanic_government
						}
					}
				}
			}
			if = {
				limit = { NOT = { has_province_modifier = imperial_cultural_tolerance } }
				add_province_modifier = {
					name = imperial_cultural_tolerance
					hidden = yes
					duration = -1
				}
			}
		}
		else_if = {
			limit = { has_province_modifier = imperial_cultural_tolerance }
			remove_province_modifier = imperial_cultural_tolerance
		}
	}
}

# If Imperial characters somehow become not Imperial, set them back

	id = 302545

	picture = "GFX_evt_riders_on_hill"
