action_clan_sentiment = {
	attribute = diplomacy
	
	potential = {
		FROM = {
			is_nomadic = yes
		}
	}
	
	trigger = {
		owner = {
			is_nomadic = yes
			NOT = { character = FROM }

			FROM = {
				job_chancellor = {
					NOT = { character = PREVPREV } # Can't be sent to improve relations with himself...
				}
			}
			
			OR = {
				vassal_of = FROM
				same_liege = FROM
				overlord_of = FROM
			}
		}
	}
	
	events = { 20235 }
}

action_promote_romanization = {
	attribute = diplomacy
	
	potential = {
		FROM = {
			government = imperial_government
			tier = EMPEROR
		}
	}
	
	trigger = {
		any_province_lord = {
			OR = {
				character = FROM
				is_liege_or_above = FROM
			}
		}
	}
	
	
	display_on_map = {
		tooltip = JOB_PROMOTE_ROMANIZATION_ROMAN_CULTURE
		trigger = {
			is_roman_culture_trigger = yes
			any_province_lord = {
				OR = {
					character = FROM
					is_liege_or_above = FROM
				}
			}
		}
		color = { 0 0 255 }
	}
	
	display_on_map = {
		tooltip = JOB_PROMOTE_ROMANIZATION_NOT_ROMAN_CULTURE
		trigger = {
			NOT = { is_roman_culture_trigger = yes }
			any_province_lord = {
				OR = {
					character = FROM
					is_liege_or_above = FROM
				}
			}
		}
		color = { 0 255 0 }
	}
	
	events = { 302220 302222 }
}

action_fabricate_claims = {
	attribute = diplomacy
	
	potential = {
		FROM = { 
			is_nomadic = no
			OR = {
				is_tribal = no
				AND = {
					government = high_tribal_government
					OR = {
						is_brythonic_culture_trigger = yes
						culture_group = imperiale
						is_romanized_germanic_trigger = yes
					}
				}
			}
			NAND = {
				government = imperial_government
				tier = EMPEROR
			}
		}
	}
	
	trigger = {
		OR = {
			owner = {
				AND = {
					religion_group = christian
					top_liege = { religion_group = christian }
					FROM = { religion_group = christian }
				}
			}
			any_province_lord = {
				NOT = { character = FROM }
				OR = {
					culture_group = FROM
					top_liege = { culture_group = FROM }
				}
			}
			FROM = {
				culture = roman
			}
			FROM = {
				any_courtier = {
					has_job_title = job_chancellor
					culture_group = ROOT
				}
			}
			FROM = {
				any_vassal = {
					has_job_title = job_chancellor
					culture_group = ROOT
				}
			}
		}
		is_within_diplo_range = FROM
	}
	
	events = { 913 914 }
}

# Lord Spiritual
action_inquisition = {
	attribute = learning
	
	potential = {
	}
	
	trigger = {
		NAND = {
			FROM = { job_spiritual = { has_character_modifier = spiritual_refused_pagans } }  # See Event ZE.5040
			owner = { religion_group = pagan_group }
		}
		any_province_lord = {
			OR = {
				character = FROM
				is_liege_or_above = FROM
			}
		}
	}
	
	display_on_map = {
		tooltip = JOB_PROSELYTIZE_SAME_RELIGION
		trigger = {
			religion = FROM
			any_province_lord = {
				OR = {
					character = FROM
					is_liege_or_above = FROM
				}
			}
		}
		color = { 0 0 255 }
	}
	
	display_on_map = {
		tooltip = JOB_PROSELYTIZE_OTHER_PAGAN_RELIGION
		trigger = {
			NOT = { religion = FROM }
			FROM = { 
				religion_group = pagan_group 
				NOT = { has_religion_feature = religion_proselytizing }
			}
			OR = {
				religion_group = pagan_group
				AND = {
					religion_group = zoroastrian_group
					culture_group = imperiale
				}
				AND = {
					religion_group = jewish_group
					culture_group = israelite
				}
			}
			NOR = {
				is_heresy_of = FROM
				FROM = {
					is_heresy_of = PREV
				}
			}
			any_province_lord = {
				OR = {
					character = FROM
					is_liege_or_above = FROM
				}
			}
		}
		color = { 0 100 255 }
	}
	
	display_on_map = {
		tooltip = JOB_PROSELYTIZE_TO_OTHER_PAGAN_RELIGION
		trigger = {
			NOT = { religion = FROM }
			FROM = { 
				religion_group = pagan_group 
				NOT = { has_religion_feature = religion_proselytizing }
			}
			FROM = { is_same_default_pagan_religion_as_ROOT = no }
			any_province_lord = {
				OR = {
					character = FROM
					is_liege_or_above = FROM
				}
			}
		}
		color = { 0 255 100 }
	}
	
	display_on_map = {
		tooltip = JOB_PROSELYTIZE_DIFFERENT_RELIGION
		trigger = {
			NOR = { 
				religion = FROM 
				AND = {
					FROM = { 
						religion_group = pagan_group
						NOT = { has_religion_feature = religion_proselytizing }
					}
					OR = {
						religion_group = pagan_group
						AND = {
							religion_group = zoroastrian_group
							culture_group = imperiale
						}
						AND = {
							religion_group = jewish_group
							culture_group = israelite
						}
					}
					NOR = {
						is_heresy_of = FROM
						FROM = {
							is_heresy_of = PREV
						}
					}
				}
			}
			any_province_lord = {
				OR = {
					character = FROM
					is_liege_or_above = FROM
				}
			}
		}
		color = { 0 255 0 }
	}
	
	events = { 900 901 902 903 }
}
