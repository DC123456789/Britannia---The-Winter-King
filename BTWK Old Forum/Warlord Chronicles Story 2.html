  3. JasperClay - 10-01-2016, 13:06
    Yes, I just figured this out last night, while fruitlessly searching for where the event target "arthur_pendragon" was set. Quite ingenious. I still don't quite have it working, because Guinevere and Ceinwyn were both taken as concubines when they couldn't marry.

    Is there anyway to just bar that? I'm currently just adding -50% fertility to forbid marriage, and having the event clear lovers, and consorts (is this a concubine/man?), and spouses. But it would be nicer just to make anything that ties these characters down impossible.
    Code:
    character_event = {	# Event for Arthur - Guinevere Marriage
	id = 312102
	desc = "EVTDESC312102"
	picture = GFX_evt_council
	
	only_rulers = yes
	has_character_flag = arthur		#Event fires for Arthur, flag set by return decision

	trigger = {
		NOT = { has_global_flag = arthur_married }
		is_married = no
		OR = {
			top_liege = {
				has_landed_title = k_dumnonia
				regent = {
					character = ROOT
				}
				character = 5 		#Either, Mordred is King and independent, and Arthur is regent
			}
			ROOT = { has_landed_title = k_dumnonia }	#or, King Arthur	 
		}
	}
	
	immediate = {
		fertility = 0.55
		remove_trait = forbid_marriage	#Arthur
		event_target:guinevere_pendragon = {
			remove_trait = forbid_marriage
			remove_spouse = yes
			remove_consort = yes
			any_lover = {
				remove_lover = yes
			}
			fertility = 0.35
		}
		event_target:derfel_cadarn = {
			remove_trait = forbid_marriage
			remove_spouse = yes
			any_lover = {
				remove_lover = yes
			}
		}
		event_target:ceinwyn_seren = {
			remove_trait = forbid_marriage
			remove_spouse = yes
			remove_consort = yes
			any_lover = {
				remove_lover = yes
			}
			fertility = 0.45
		}
		event_target:lancelot_dulac = {
			remove_trait = forbid_marriage
			remove_spouse = yes
			remove_consort = yes
			any_lover = {
				remove_lover = yes
			}
		}
	}
	
	option = {
		name = "EVTOPTA312102"
		ai_chance = {
			factor = 95
		}
		add_spouse = 26		#Guinevere
		add_lover = 26		#Guinevere
		event_target:target_ceinwyn = {
			add_spouse = 10
			add_lover = 10
		}
		set_global_flag = arthur_married
	}
	
	option = {
		name = "EVTOPTB312102"
		ai_chance = {
			factor = 5
		}
		add_spouse = 27 	#Ceinwyn
		event_target:lancelot_dulac = { 
			add_spouse = 26 	#Guinevere
			add_lover = 26		#Guinevere
		}
		any_playable_ruler = {
			any_courtier = {
				limit = { character = 6 }	#Nimue
				remove_lover = 2 	#Merlin
				add_lover = 10 		#Derfel
			}
		}
		set_global_flag = arthur_married
	}
}
